Configure as tabelas no Supabase
Execute este SQL no seu projeto Supabase para ativar os investimentos:
CREATE TABLE investment_accounts (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users ON DELETE CASCADE, name text NOT NULL, institution text, type text DEFAULT 'corretora', color text DEFAULT 'var(--accent-2)', created_at timestamptz DEFAULT now()); CREATE TABLE investments (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users ON DELETE CASCADE, account_id uuid REFERENCES investment_accounts ON DELETE SET NULL, name text NOT NULL, symbol text, type text NOT NULL, quantity numeric, purchase_price numeric DEFAULT 0, purchase_date date, rate_pct numeric, rate_index text, notes text, created_at timestamptz DEFAULT now()); CREATE TABLE investment_dividends (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users ON DELETE CASCADE, investment_id uuid REFERENCES investments ON DELETE CASCADE, account_id uuid REFERENCES investment_accounts ON DELETE SET NULL, type text DEFAULT 'dividendo', value_per_share numeric DEFAULT 0, payment_date date, created_at timestamptz DEFAULT now()); CREATE TABLE investment_account_events (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users ON DELETE CASCADE, account_id uuid REFERENCES investment_accounts ON DELETE CASCADE, type text NOT NULL, amount numeric NOT NULL, description text, event_date date DEFAULT CURRENT_DATE, reference_id uuid, created_at timestamptz DEFAULT now()); ALTER TABLE investment_accounts ENABLE ROW LEVEL SECURITY; ALTER TABLE investments ENABLE ROW LEVEL SECURITY; ALTER TABLE investment_dividends ENABLE ROW LEVEL SECURITY; ALTER TABLE investment_account_events ENABLE ROW LEVEL SECURITY; CREATE POLICY "own" ON investment_accounts FOR ALL USING (auth.uid()=user_id); CREATE POLICY "own" ON investments FOR ALL USING (auth.uid()=user_id); CREATE POLICY "own" ON investment_dividends FOR ALL USING (auth.uid()=user_id); CREATE POLICY "own" ON investment_account_events FOR ALL USING (auth.uid()=user_id);
